5. Adding elements into array

       

           push()            It adds new element(s) as last item(s).

           unshift()        It adds new element(s) as first item(s).

           splice()            It adds new element(s) at specific position. 



   Syntax:

           collection.push(“item1”, “item2”, …)

           collection.unshift(“item1”, “item2”, …)

           collection.splice(startIndex, deleteCount, “item1”, “item2”, …)

           [delete count must be zero, if you want to add without removing existing]



Ex:

<script>

     var categories = ["Fashion", "Footwear"];



     categories.push("Electronics");

     categories.unshift("All");

     categories.splice(2,0,"Accessories");



     categories.map(function(value, index){

         document.write(`[${index}] : ${value}<br>`);

     })

    

</script>





6. Removing elements from array



           pop()                It removes and returns the last item.

           shift()            It removes and returns the first item.

           splice()            It removes and returns specific item(s).



Syntax:

           collection.pop();

           collection.shift();

           collection.splice(startIndex, deleteCount)



Ex:

<script>

     var categories = ["Fashion", "Footwear"];





     function PrintCollection(){

            categories.map(function(value, index){

            document.write(`[${index}] : ${value}<br>`);

        })

     }



     categories.push("Electronics");

     categories.unshift("All");

     categories.splice(2,0,"Accessories");



     PrintCollection();

     

     document.write("<hr>");



     console.log(categories.pop());

     console.log(categories.shift());

     console.log(categories.splice(1,1));

     PrintCollection();

    

</script>



7. Sorting array elements



       sort()                It arranges elements in ascending order.

                           It uses a string format for sorting. 



       reverse()        It arranges elements in reverse order. 



Ex:

<script>

     var categories = ["D", "C", "A", "B"];





     function PrintCollection(){

            categories.map(function(value, index){

            document.write(`[${index}] : ${value}<br>`);

        })

     }

     categories.sort().reverse();



     PrintCollection();

    

</script>



Dynamically presenting elements in UI:

   

   1. Create element dynamically by using document method “createElement()”.



               document.createElement(“img”)

               document.createElement(“ol”)

               document.createElement(“input”) … 



   2. Assign a memory reference for elements in dynamic memory. 



               var  pic  = document.createElement(“img”);

               var  input = document.createElement(“input”);



   3. Set properties for elements



               pic.src = “ ”;

               pic.alt = “ ”;

               

               input.type = “password | number | range | checkbox..”;



   4. Add element into UI by using the document method “appendChild()”.



               appendChild()        It is used to add a child element into specific parent. 



               document.querySelector(“body”).appendChild(pic);

               document.querySelector(“form”).appendChild(input);
