2. prompt()

       - It is an input box of browser window. 

       - It allows user to input a value. 



Syntax:

            prompt(“message”, “default_value”);         // default value is optional 



       - Prompt returns the following 



                  null            :    on cancel

                  ‘  ’            :    empty string on OK without value

                  ‘value’    :    returns value on OK with value



Ex:

<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Document</title>

    <script>

        function SearchClick(){

          result =  prompt("Search Products", "eg: shoes, jeans");

          if(result==null) {

              alert("You canceled search");

          }

          else if(result=='') {

                alert('Please provide a product name');

          } else {

              document.querySelector("p").textContent = "You are searching for :" + result;

          }

        }

    </script>

</head>

<body>

    <button onclick="SearchClick()">Search</button>

    <p></p>

</body>

</html>





3. Form Input Elements

       - HTML form provides various input elements like textbox, password, checkbox, radio

         dropdown, listbox etc.

       - JavaScript can access input value from form elements and use in application. 



Syntax:

               <input type=“text”  id=“userName”>

               <select  id=“list”> 

                 <option> </option>

               </select>



               document.getElementById(“userName”).value 

               document.getElementById(“list”).value

function BookClick(){

     document.getElementById("summaryContainer").style.display = "block";

     document.getElementById("buttonContainer").style.display = "none";



     document.getElementById("lblMovie").textContent = document.getElementById("lstMovies").value;

     document.getElementById("lblDate").textContent = document.getElementById("lstDate").value;

     document.getElementById("lblCinema").textContent = document.getElementById("lstCinema").value;

     document.getElementById("lblTiming").textContent = document.getElementById("lstTiming").value; 



     poster = document.getElementById("imgPoster");

     movieName = document.getElementById("lstMovies").value; 



     if(movieName=="THAMMA") {

         poster.src = "../public/images/thamma.png";

     } else {

        poster.src = "../public/images/llb.png"

     }



}

function ModifyClick(){

    document.getElementById("lblBooking").textContent = "Modify Booking";

    document.getElementById("btnBook").textContent = "Save";

    document.getElementById("btnBook").className = "btn btn-success";

}
